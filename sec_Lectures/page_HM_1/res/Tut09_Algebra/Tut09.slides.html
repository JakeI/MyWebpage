def view_complex(z, c, t, dx, dy):
    xy = (np.real(z), np.imag(z))
    ax.annotate("", xy=xy, 
                xycoords='data', xytext=(0, 0), 
                textcoords='data',
                arrowprops=dict(arrowstyle="->",
                                connectionstyle="arc3",
                                color=c)
                )
    ax.text(s=t, x=xy[0]+dx, y=xy[1]+dy, color=c)

view_complex(np.exp(-1j*np.pi/4), "red", "$z^3$", 0, 0)
view_complex(np.exp(1j*5*np.pi/4), "blue", "$z$", 0, -0.03)
view_complex(-1j, "green", "$z^{150}$", 0.02, 0)
